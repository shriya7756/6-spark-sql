cd /home/hadoop/spark-3.5.2-bin-hadoop3
cd bin
./spark-sql
CREATE TABLE flights (
DEST_COUNTRY_NAME STRING,
ORIGIN_COUNTRY_NAME STRING,
count LONG
)
USING JSON
OPTIONS (path '/home/hadoop/Desktop/SparkSQLrmm/2015-summary.json');

DESCRIBE TABLE flights;

SELECT * FROM flights;

SELECT
CASE
WHEN DEST_COUNTRY_NAME = 'UNITED STATES' THEN 1
WHEN DEST_COUNTRY_NAME = 'Egypt' THEN 0
ELSE -1
END
FROM flights;

SELECT DEST_COUNTRY_NAME, ARRAY(1, 2, 3)
FROM flights;

SELECT DEST_COUNTRY_NAME AS new_name,
collect_list(count)[0]
FROM flights
GROUP BY DEST_COUNTRY_NAME;

SHOW FUNCTIONS;

SHOW SYSTEM FUNCTIONS;

SHOW USER FUNCTIONS;

SHOW FUNCTIONS "s*";

SHOW FUNCTIONS LIKE "collect*";

SELECT dest_country_name
FROM flights
GROUP BY dest_country_name
ORDER BY sum(count) DESC
LIMIT 5;

SELECT *
FROM flights
WHERE origin_country_name IN (
SELECT dest_country_name
FROM flights
GROUP BY dest_country_name
ORDER BY sum(count) DESC
LIMIT 5
);

SELECT *
FROM flights f1
WHERE EXISTS (
SELECT 1 FROM flights f2
WHERE f1.dest_country_name = f2.origin_country_name
)
AND EXISTS (
SELECT 1 FROM flights f2
WHERE f2.dest_country_name = f1.origin_country_name
);
